def clear
  @lock.synchronize do
    Remotus.logger.debug { "Removing all host pools" }
    return 0 unless @pool

    num_pools = count

    # Force connection close
    @pool.each { |_hostname, host_pool| host_pool.close }
    @pool = {}

    return num_pools
  end
end
